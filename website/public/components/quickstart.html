Get started in 60 seconds

1

Install

npm install spark-html
2

Add the Vite plugin & mount

// vite.config.js
import spark from 'spark-html/vite';
export default { plugins: [spark()] };

// main.js
import { mount } from 'spark-html';
mount();
3

Write a component & import it

<!-- components/hello.html -->
<h1>Hello {planet}</h1>
<script> let planet = 'world'; </script>

<!-- index.html -->
<div import="components/hello"></div>
